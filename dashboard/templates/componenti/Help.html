/registrazione_azienda

        WINDOWS ESEMPIO:
        
            $azienda = @{
                Ragionesociale = ""
                Partitaiva = ""
                Indirizzo = ""
                Comune = ""
                Provincia = ""
                Cap = ""
                Nazione = ""
                Email = ""
                Pec = ""
                Tel = "" 
            }
            
            $jsonAzienda = $azienda | ConvertTo-Json
            
            $url = "???/registrazione_azienda"
            Invoke-RestMethod -Uri $url -Method Post -Body $jsonAzienda -ContentType "application/json"
        



        LINUX ESEMPIO (con cURL):
        
            #!/bin/bash

            azienda='{
                "Ragionesociale": "",
                "Partitaiva": "",
                "Indirizzo": "",
                "Comune": "",
                "Provincia": "",
                "Cap": "",

                "Email": "",
                "Pec": "",
                "Tel": ""
            }'

            url="http://???/registrazione_azienda"
            curl -X POST -d "$azienda" -H "Content-Type: application/json" "$url"
